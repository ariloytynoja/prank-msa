HSAML format

An alignment generated by PRANK consist of four elements: newick, nodes, selection and model. A sub-element of nodes is either leaf or node.

<ms_alignment>
<newick>
((seq1:0.100000,seq2:0.200000)#1#:0.020000,seq3:0.150000)#2#
</newick>
<nodes>
<leaf id="seq1" name="human">
<sequence>
GTGCCACCCA
</sequence>
</leaf>
<leaf id="seq2" name="mouse">
<sequence>
GTACCAGCCT
</sequence>
</leaf>
<leaf id="seq3" name="dog">
<sequence>
ATGCGA-CCA
</sequence>
</leaf>
<node id="#1#">
<probability id="1">
51,54,56,58,59,61,63,65,66,68
</probability>
<probability id="2">
49,46,44,42,41,39,37,35,34,32
</probability>
<probability id="3">
100,100,100,100,100,100,100,100,100,100
</probability>
</node>
<node id="#2#">
<probability id="1">
54,55,57,58,60,60,61,62,64,66
</probability>
<probability id="2"> 46,45,43,42,40,40,39,38,36,34
</probability>
<probability id="3">
100,100,99,93,87,84,35,52,75,98
</probability>
</node>
</nodes>
<selection>
<selected_sites>
1,1,1,1,1,1,0,0,0,1
</selected_sites>
<selection_criteria>
select all;
any node: unselect pprob less than 75% between 0-10;
</selection_criteria>
</selection>
<model>
<probability id="1" name="fast" color="green" style="bar" show="yes"/>
<probability id="2" name="slow" color="red" style="bar" show="yes"/>
<probability id="3" name="pprob" color="gray" style="bar" show="yes"/>
</model>
</ms_alignment>
